            <!-- an example to test the POST call -->
            <button id="test_add_AAP">test add_AAP()</button>
            <script>
                document.getElementById("test_add_AAP").addEventListener("click", function () {
                    // Data to be sent in the POST request
                    const aapData = {
                        name: "aap stuff",
                        description: "This is an aap",
                    };

                    // Send a POST request to your Django server
                    fetch('/instructor/AAP/', {
                        method: 'POST',
                        headers: {
                            'Content-Type': 'application/json', // Set the content type to JSON
                        },
                        body: JSON.stringify(aapData), // Convert the data to JSON string
                    })
                    .then(response => {
                        if (response.ok) {
                            return response.json(); // Parse the response JSON if successful
                        }
                        throw new Error('Network response was not ok');
                    })
                    .then(data => {
                        console.log(data); // Process the response data
                        // You can update the DOM or perform other actions here
                    })
                    .catch(error => {
                        console.error('There was a problem with the fetch operation:', error);
                    });
                });
            </script>
